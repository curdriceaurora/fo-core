```python
"""Timestamp safety tests.

Verify that the codebase does not leak naive datetimes from public APIs
and that timestamp-dependent operations use timezone-aware datetimes.
"""
import time_machine
from datetime import UTC, datetime, timezone

class TestTimestampSafety:
    """Verify timezone-aware datetime usage across modules."""

    def test_fromtimestamp_produces_aware_datetime(self):
        """fromtimestamp with tz=UTC produces timezone-aware result."""
        ts = 1709000000.0
        dt = datetime.fromtimestamp(ts, tz=UTC)
        assert dt.tzinfo is not None

    def test_backup_timestamp_has_microseconds(self):
        """Backup timestamp format includes microseconds for uniqueness."""
        fmt = datetime.now(UTC).strftime('%Y%m%d_%H%M%S_%f')
        parts = fmt.split('_')
        assert len(parts) == 3
        assert len(parts[2]) == 6  # microseconds

    @time_machine.travel("2026-03-01 12:00:00", tick=False)
    def test_frozen_time_is_consistent(self):
        """Frozen time produces consistent timestamps."""
        t1 = datetime.now(UTC)
        t2 = datetime.now(UTC)
        assert t1 == t2

    def test_conflict_resolver_uses_aware_datetimes(self):
        """ConflictResolver handles timestamps without naive/aware crash."""
        from file_organizer.services.intelligence.conflict_resolver import ConflictResolver
        resolver = ConflictResolver()
        prefs = [
            {"updated": "2026-01-01T00:00:00Z", "correction_count": 5, "confidence": 0.8},
            {"updated": "2026-02-01T00:00:00Z", "correction_count": 10, "confidence": 0.9},
        ]
        result = resolver.resolve(prefs)
        assert result is not None

    def test_operation_history_timestamp_format(self):
        """Operation history timestamps use Z suffix, not +00:00Z."""
        ts = datetime.now(UTC).isoformat().replace("+00:00", "Z")
        assert ts.endswith("Z")
        assert "+00:00Z" not in ts
```

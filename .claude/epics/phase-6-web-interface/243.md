```python
# file_organizer_client/client.py

import requests
from typing import List, Dict, Any, Optional
from pathlib import Path
from .models import OrganizeResult, FileMetadata, DuplicateGroup
from .exceptions import AuthenticationError, APIError

class FileOrganizerClient:
    """File Organizer API client."""

    def __init__(
        self,
        base_url: str = "http://localhost:8000",
        api_key: Optional[str] = None,
        timeout: int = 30
    ):
        """
        Initialize the client.

        Args:
            base_url: Base URL of File Organizer API
            api_key: API key for authentication
            timeout: Request timeout in seconds
        """
        self.base_url = base_url.rstrip("/")
        self.api_key = api_key
        self.timeout = timeout
        self.session = requests.Session()

        if api_key:
            self.session.headers.update({"Authorization": f"Bearer {api_key}"})

    def organize(
        self,
        path: str,
        method: str = "para",
        dry_run: bool = False,
        **kwargs
    ) -> OrganizeResult:
        """
        Organize files.

        Args:
            path: Path to organize
            method: Organization method (para, johnny_decimal, etc.)
            dry_run: Preview changes without applying
            **kwargs: Additional method-specific options

        Returns:
            OrganizeResult with operation details

        Raises:
            APIError: If the API request fails
        """
        response = self._post("/api/organize", {
            "path": path,
            "method": method,
            "dry_run": dry_run,
            **kwargs
        })
        return OrganizeResult.from_dict(response)

    def get_metadata(self, path: str) -> FileMetadata:
        """
        Get file metadata.

        Args:
            path: File path

        Returns:
            FileMetadata object

        Raises:
            APIError: If the API request fails
        """
        response = self._get(f"/api/metadata", params={"path": path})
        return FileMetadata.from_dict(response)

    def find_duplicates(
        self,
        path: str,
        method: str = "hash",
        min_size: Optional[int] = None
    ) -> List[DuplicateGroup]:
        """
        Find duplicate files.

        Args:
            path: Directory to scan
            method: Deduplication method (hash, perceptual, etc.)
            min_size: Minimum file size in bytes

        Returns:
            List of duplicate groups

        Raises:
            APIError: If the API request fails
        """
        params = {"path": path, "method": method}
        if min_size:
            params["min_size"] = min_size

        response = self._get("/api/duplicates", params=params)
        return [DuplicateGroup.from_dict(g) for g in response["groups"]]

    def get_config(self) -> Dict[str, Any]:
        """
        Get current configuration.

        Returns:
            Configuration dictionary

        Raises:
            APIError: If the API request fails
        """
        return self._get("/api/config")

    def update_config(self, config: Dict[str, Any]) -> Dict[str, str]:
        """
        Update configuration.

        Args:
            config: Configuration updates

        Returns:
            Status message

        Raises:
            APIError: If the API request fails
        """
        return self._put("/api/config", config)

    def health_check(self) -> Dict[str, Any]:
        """
        Check API health.

        Returns:
            Health status

        Raises:
            APIError: If the API request fails
        """
        return self._get("/health")

    # Private methods

    def _get(self, path: str, params: Optional[Dict] = None) -> Any:
        """Make GET request."""
        return self._request("GET", path, params=params)

    def _post(self, path: str, data: Dict) -> Any:
        """Make POST request."""
        return self._request("POST", path, json=data)

    def _put(self, path: str, data: Dict) -> Any:
        """Make PUT request."""
        return self._request("PUT", path, json=data)

    def _delete(self, path: str) -> Any:
        """Make DELETE request."""
        return self._request("DELETE", path)

    def _request(
        self,
        method: str,
        path: str,
        **kwargs
    ) -> Any:
        """Make HTTP request."""
        url = f"{self.base_url}{path}"

        try:
            response = self.session.request(
                method,
                url,
                timeout=self.timeout,
                **kwargs
            )
            response.raise_for_status()
            return response.json()
        except requests.exceptions.HTTPError as e:
            if e.response.status_code == 401:
                raise AuthenticationError("Invalid API key")
            raise APIError(f"API request failed: {e}")
        except requests.exceptions.RequestException as e:
            raise APIError(f"Request failed: {e}")
```

```python
from datetime import datetime

@dataclass

    """File metadata."""

    created: datetime
    modified: datetime
    mime_type: str

    tags: List[str] = None

        return cls(
            path=data["path"],
            size=data["size"],
            created=datetime.fromisoformat(data["created"]),
            modified=datetime.fromisoformat(data["modified"]),
            mime_type=data["mime_type"],
```

```python
class AsyncFileOrganizerClient:
    """Async File Organizer API client."""

    def __init__(
        self,
        base_url: str = "http://localhost:8000",
        api_key: Optional[str] = None,
        timeout: int = 30
    ):
        """Initialize async client."""
        self.base_url = base_url.rstrip("/")
        self.api_key = api_key
        self.timeout = aiohttp.ClientTimeout(total=timeout)
        self.headers = {}

        if api_key:
            self.headers["Authorization"] = f"Bearer {api_key}"

    async def organize(
        self,
        path: str,
        method: str = "para",
        dry_run: bool = False,
        **kwargs
    ) -> OrganizeResult:
        """Organize files asynchronously."""
        async with aiohttp.ClientSession(headers=self.headers) as session:
            async with session.post(
                f"{self.base_url}/api/organize",
                json={
                    "path": path,
                    "method": method,
                    "dry_run": dry_run,
                    **kwargs
                },
                timeout=self.timeout
            ) as response:
                response.raise_for_status()
                data = await response.json()
                return OrganizeResult.from_dict(data)
```

```typescript
// file-organizer-client/src/client.ts

import axios, { AxiosInstance, AxiosRequestConfig } from 'axios';
import {
  OrganizeOptions,
  OrganizeResult,
  FileMetadata,
  DuplicateGroup,
  HealthStatus
} from './types';
import { FileOrganizerError, AuthenticationError, APIError } from './errors';

export class FileOrganizerClient {
  private client: AxiosInstance;

  /**
   * Create a new File Organizer client
   *
   * @param baseUrl - Base URL of File Organizer API
   * @param apiKey - API key for authentication
   * @param timeout - Request timeout in milliseconds
   */
  constructor(
    baseUrl: string = 'http://localhost:8000',
    apiKey?: string,
    timeout: number = 30000
  ) {
    this.client = axios.create({
      baseURL: baseUrl,
      timeout: timeout,
      headers: apiKey ? { Authorization: `Bearer ${apiKey}` } : {}
    });

    // Add response interceptor for error handling
    this.client.interceptors.response.use(
      response => response,
      error => {
        if (error.response?.status === 401) {
          throw new AuthenticationError('Invalid API key');
        }
        throw new APIError(error.message);
      }
    );
  }

  /**
   * Organize files
   *
   * @param path - Path to organize
   * @param options - Organization options
   * @returns Organization result
   */
  async organize(path: string, options: OrganizeOptions = {}): Promise<OrganizeResult> {
    const response = await this.client.post<OrganizeResult>('/api/organize', {
      path,
      method: options.method || 'para',
      dry_run: options.dryRun || false,
      ...options
    });
    return response.data;
  }

  /**
   * Get file metadata
   *
   * @param path - File path
   * @returns File metadata
   */
  async getMetadata(path: string): Promise<FileMetadata> {
    const response = await this.client.get<FileMetadata>('/api/metadata', {
      params: { path }
    });
    return response.data;
  }

  /**
   * Find duplicate files
   *
   * @param path - Directory to scan
   * @param method - Deduplication method
   * @param minSize - Minimum file size
   * @returns List of duplicate groups
   */
  async findDuplicates(
    path: string,
    method: string = 'hash',
    minSize?: number
  ): Promise<DuplicateGroup[]> {
    const params: any = { path, method };
    if (minSize) params.min_size = minSize;

    const response = await this.client.get<{ groups: DuplicateGroup[] }>(
      '/api/duplicates',
      { params }
    );
    return response.data.groups;
  }

  /**
   * Get configuration
   *
   * @returns Configuration object
   */
  async getConfig(): Promise<Record<string, any>> {
    const response = await this.client.get('/api/config');
    return response.data;
  }

  /**
   * Update configuration
   *
   * @param config - Configuration updates
   * @returns Status message
   */
  async updateConfig(config: Record<string, any>): Promise<{ status: string }> {
    const response = await this.client.put('/api/config', config);
    return response.data;
  }

  /**
   * Check API health
   *
   * @returns Health status
   */
  async healthCheck(): Promise<HealthStatus> {
    const response = await this.client.get<HealthStatus>('/health');
    return response.data;
  }
}
```

### CLI Wrapper

```python
# file_organizer_client/cli.py

import typer
from pathlib import Path
from .client import FileOrganizerClient
from rich.console import Console
from rich.table import Table

app = typer.Typer(help="File Organizer API CLI")
console = Console()

@app.command()
def organize(
    path: Path = typer.Argument(..., help="Path to organize"),
    method: str = typer.Option("para", "--method", "-m", help="Organization method"),
    dry_run: bool = typer.Option(False, "--dry-run", "-d", help="Preview changes"),
    api_key: str = typer.Option(None, envvar="FILE_ORGANIZER_API_KEY")
):
    """Organize files via API."""
    client = FileOrganizerClient(api_key=api_key)
    result = client.organize(str(path), method=method, dry_run=dry_run)

    if result.success:
        console.print(f"[green]✓[/green] Organized {result.files_moved} files")
    else:
        console.print("[red]✗[/red] Organization failed")
        for error in result.errors:
            console.print(f"  [red]•[/red] {error}")

@app.command()
def duplicates(
    path: Path = typer.Argument(..., help="Path to scan"),
    method: str = typer.Option("hash", "--method", "-m", help="Detection method"),
    api_key: str = typer.Option(None, envvar="FILE_ORGANIZER_API_KEY")
):
    """Find duplicate files via API."""
    client = FileOrganizerClient(api_key=api_key)
    groups = client.find_duplicates(str(path), method=method)

    table = Table(title="Duplicate Files")
    table.add_column("Hash", style="cyan")
    table.add_column("Size", style="magenta")
    table.add_column("Count", style="green")
    table.add_column("Files", style="yellow")

    for group in groups:
        table.add_row(
            group.hash[:8],
            f"{group.size:,}",
            str(group.count),
            "
".join(group.files)
        )

    console.print(table)

@app.command()
def metadata(
    path: Path = typer.Argument(..., help="File path"),
    api_key: str = typer.Option(None, envvar="FILE_ORGANIZER_API_KEY")
):
    """Get file metadata via API."""
    client = FileOrganizerClient(api_key=api_key)
    metadata = client.get_metadata(str(path))

    console.print(f"[bold]File:[/bold] {metadata.path}")
    console.print(f"[bold]Size:[/bold] {metadata.size:,} bytes")
    console.print(f"[bold]Type:[/bold] {metadata.mime_type}")
    console.print(f"[bold]Created:[/bold] {metadata.created}")
    console.print(f"[bold]Modified:[/bold] {metadata.modified}")
    if metadata.hash:
        console.print(f"[bold]Hash:[/bold] {metadata.hash}")

if __name__ == "__main__":
    app()
```

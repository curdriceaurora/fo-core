### Hooks and Events System

```python
# file_organizer/plugins/api/hooks.py

from typing import Callable, Any, Dict, List
from enum import Enum

class HookEvent(Enum):
    """Plugin hook events."""
    # File events
    FILE_SCANNED = "file.scanned"
    FILE_ORGANIZED = "file.organized"
    FILE_DUPLICATED = "file.duplicated"
    FILE_DELETED = "file.deleted"

    # Organization events
    ORG_STARTED = "organization.started"
    ORG_COMPLETED = "organization.completed"
    ORG_FAILED = "organization.failed"

    # Deduplication events
    DEDUPE_STARTED = "deduplication.started"
    DEDUPE_COMPLETED = "deduplication.completed"
    DEDUPE_FOUND = "deduplication.found"

    # Methodology events
    PARA_CATEGORIZED = "para.categorized"
    JD_ASSIGNED = "johnny_decimal.assigned"

class Hook:
    """Hook definition."""

    def __init__(self, event: HookEvent, callback: Callable):
        self.event = event
        self.callback = callback
        self.priority = 10  # Default priority

    async def execute(self, context: Dict[str, Any]) -> Any:
        """Execute hook callback."""
        return await self.callback(context)

class HookManager:
    """Manages plugin hooks."""

    def __init__(self):
        self._hooks: Dict[HookEvent, List[Hook]] = {}

    def register(self, event: HookEvent, callback: Callable, priority: int = 10) -> None:
        """Register a hook."""
        hook = Hook(event, callback)
        hook.priority = priority

        if event not in self._hooks:
            self._hooks[event] = []

        self._hooks[event].append(hook)
        self._hooks[event].sort(key=lambda h: h.priority)

    def unregister(self, event: HookEvent, callback: Callable) -> None:
        """Unregister a hook."""
        if event in self._hooks:
            self._hooks[event] = [h for h in self._hooks[event] if h.callback != callback]

    async def trigger(self, event: HookEvent, context: Dict[str, Any]) -> List[Any]:
        """Trigger all hooks for an event."""
        results = []
        if event in self._hooks:
            for hook in self._hooks[event]:
                try:
                    result = await hook.execute(context)
                    results.append(result)
                except Exception as e:
                    # Log error but continue with other hooks
                    print(f"Hook execution failed: {e}")
        return results

    def list_hooks(self, event: Optional[HookEvent] = None) -> List[Hook]:
        """List registered hooks."""
        if event:
            return self._hooks.get(event, [])
        return [hook for hooks in self._hooks.values() for hook in hooks]
```

```python
# file_organizer/plugins/sdk/client.py

import requests
from typing import Any, Dict, List, Optional
from pathlib import Path

class PluginClient:
    """Client for interacting with File Organizer API."""

    def __init__(self, base_url: str, token: str):
        self.base_url = base_url
        self.token = token
        self.session = requests.Session()
        self.session.headers.update({"Authorization": f"Bearer {token}"})

    def list_files(self, path: str, recursive: bool = False) -> List[Dict[str, Any]]:
        """List files."""
        response = self.session.get(
            f"{self.base_url}/api/plugin/files/list",
            params={"path": path, "recursive": recursive}
        )
        response.raise_for_status()
        return response.json()

    def get_metadata(self, path: str) -> Dict[str, Any]:
        """Get file metadata."""
        response = self.session.get(
            f"{self.base_url}/api/plugin/files/metadata",
            params={"path": path}
        )
        response.raise_for_status()
        return response.json()

    def organize_file(self, path: str, destination: str) -> Dict[str, Any]:
        """Organize a file."""
        response = self.session.post(
            f"{self.base_url}/api/plugin/files/organize",
            json={"path": path, "destination": destination}
        )
        response.raise_for_status()
        return response.json()

    def get_config(self, key: str) -> Any:
        """Get configuration value."""
        response = self.session.get(
            f"{self.base_url}/api/plugin/config/get",
            params={"key": key}
        )
        response.raise_for_status()
        return response.json()

    def register_hook(self, event: str, callback_url: str) -> Dict[str, str]:
        """Register a webhook."""
        response = self.session.post(
            f"{self.base_url}/api/plugin/hooks/register",
            json={"hook_name": event, "callback_url": callback_url}
        )
        response.raise_for_status()
        return response.json()
```

```python
# file_organizer/plugins/sdk/decorators.py

from functools import wraps
from typing import Callable

def hook(event: str, priority: int = 10):
    """Decorator for hook functions."""
    def decorator(func: Callable) -> Callable:
        @wraps(func)
        async def wrapper(*args, **kwargs):
            return await func(*args, **kwargs)

        wrapper._is_hook = True
        wrapper._hook_event = event
        wrapper._hook_priority = priority
        return wrapper
    return decorator

def command(name: str, description: str = ""):
    """Decorator for plugin commands."""
    def decorator(func: Callable) -> Callable:
        @wraps(func)
        def wrapper(*args, **kwargs):
            return func(*args, **kwargs)

        wrapper._is_command = True
        wrapper._command_name = name
        wrapper._command_description = description
        return wrapper
    return decorator
```

```python
import unittest
from typing import Any, Dict
from pathlib import Path

class PluginTestCase(unittest.TestCase):
    """Base test case for plugins."""

    def setUp(self):
        """Set up test environment."""
        self.test_dir = Path("/tmp/plugin-test")
        self.test_dir.mkdir(exist_ok=True)

    def tearDown(self):
        """Clean up test environment."""
        import shutil
        if self.test_dir.exists():
            shutil.rmtree(self.test_dir)

    def create_test_file(self, name: str, content: str = "") -> Path:
        """Create a test file."""
        file_path = self.test_dir / name
        file_path.write_text(content)
        return file_path

    def assertFileExists(self, path: Path):
        """Assert file exists."""
        self.assertTrue(path.exists(), f"File does not exist: {path}")

    def assertFileNotExists(self, path: Path):
        """Assert file does not exist."""
        self.assertFalse(path.exists(), f"File exists: {path}")
```

```python
# examples/plugins/file_logger/plugin.py

from file_organizer.plugins.base import Plugin, PluginMetadata
from file_organizer.plugins.sdk import hook
from pathlib import Path

class FileLoggerPlugin(Plugin):
    """Logs file organization events."""

    def get_metadata(self) -> PluginMetadata:
        return PluginMetadata(
            name="file-logger",
            version="1.0.0",
            author="File Organizer Team",
            description="Logs all file organization events to a file"
        )

    def on_enable(self):
        self.log_file = Path(self.config.get("log_path", "file_organizer.log"))

    @hook("file.organized", priority=10)
    async def on_file_organized(self, context):
        """Log when a file is organized."""
        with open(self.log_file, "a") as f:
            f.write(f"Organized: {context['file_path']} -> {context['destination']}
")

    @hook("file.duplicated", priority=10)
    async def on_file_duplicated(self, context):
        """Log when a duplicate is found."""
        with open(self.log_file, "a") as f:
            f.write(f"Duplicate found: {context['file_path']}
")
```

#### Example 3: Auto-Backup (SDK Usage)

```python
# examples/plugins/auto_backup/plugin.py

from file_organizer.plugins.base import Plugin, PluginMetadata
from file_organizer.plugins.sdk import PluginClient, hook
import shutil
from pathlib import Path

class AutoBackupPlugin(Plugin):
    """Automatically backs up organized files."""

    def get_metadata(self) -> PluginMetadata:
        return PluginMetadata(
            name="auto-backup",
            version="1.0.0",
            author="File Organizer Team",
            description="Automatically backs up files after organization"
        )

    def on_enable(self):
        self.backup_dir = Path(self.config.get("backup_dir", "~/file-organizer-backup"))
        self.backup_dir.mkdir(parents=True, exist_ok=True)

        # Initialize SDK client
        self.client = PluginClient(
            base_url=self.config.get("api_url", "http://localhost:8000"),
            token=self.config.get("api_token")
        )

    @hook("file.organized", priority=5)
    async def on_file_organized(self, context):
        """Backup file after organization."""
        source = Path(context["destination"])
        dest = self.backup_dir / source.name

        # Create backup
        shutil.copy2(source, dest)
        print(f"Backed up: {source} -> {dest}")
```

#### Example 4: Metadata Enricher (Advanced)

```python
# examples/plugins/metadata_enricher/plugin.py

from file_organizer.plugins.base import Plugin, PluginMetadata
from file_organizer.plugins.sdk import PluginClient, hook
from pathlib import Path
import requests

class MetadataEnricherPlugin(Plugin):
    """Enriches file metadata from external sources."""

    def get_metadata(self) -> PluginMetadata:
        return PluginMetadata(
            name="metadata-enricher",
            version="1.0.0",
            author="File Organizer Team",
            description="Enriches file metadata from external APIs"
        )

    def on_enable(self):
        self.client = PluginClient(
            base_url=self.config.get("api_url"),
            token=self.config.get("api_token")
        )
        self.external_api = self.config.get("external_api_url")

    @hook("file.scanned", priority=15)
    async def on_file_scanned(self, context):
        """Enrich metadata when file is scanned."""
        file_path = Path(context["file_path"])

        # Get current metadata
        metadata = self.client.get_metadata(str(file_path))

        # Fetch additional metadata from external source
        if metadata.get("title"):
            external_data = self._fetch_external_metadata(metadata["title"])
            if external_data:
                # Update metadata
                metadata.update(external_data)
                # Store back (requires API endpoint)

    def _fetch_external_metadata(self, title: str):
        """Fetch metadata from external API."""
        try:
            response = requests.get(f"{self.external_api}/search?q={title}")
            response.raise_for_status()
            return response.json()
        except Exception as e:
            print(f"Failed to fetch external metadata: {e}")
            return None
```

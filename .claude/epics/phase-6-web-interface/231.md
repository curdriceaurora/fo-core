# Task 003: Add WebSocket Support for Real-Time Updates

## Description

Implement WebSocket support to enable real-time bidirectional communication between the backend and frontend. This allows streaming progress updates for long-running operations (scanning, organizing, deduplication), multi-client broadcast support, heartbeat/reconnection logic, and efficient event-driven updates.

WebSocket connections will complement the REST API by providing live status updates without polling.

### WebSocket Endpoint

```python
from fastapi import WebSocket, WebSocketDisconnect
from typing import Dict, Set
import asyncio
import json

class ConnectionManager:
    def __init__(self):
        self.active_connections: Dict[str, Set[WebSocket]] = {}
        self.client_subscriptions: Dict[WebSocket, Set[str]] = {}

    async def connect(self, websocket: WebSocket, client_id: str):
        await websocket.accept()
        if client_id not in self.active_connections:
            self.active_connections[client_id] = set()
        self.active_connections[client_id].add(websocket)
        self.client_subscriptions[websocket] = set()
        await self.send_personal_message(
            {"type": "connection", "status": "connected", "client_id": client_id},
            websocket
        )

    def disconnect(self, websocket: WebSocket, client_id: str):
        self.active_connections[client_id].discard(websocket)
        if not self.active_connections[client_id]:
            del self.active_connections[client_id]
        if websocket in self.client_subscriptions:
            del self.client_subscriptions[websocket]

    async def send_personal_message(self, message: dict, websocket: WebSocket):
        await websocket.send_json(message)

    async def broadcast(self, message: dict, channel: str = "global"):
        """Broadcast to all clients subscribed to a channel"""
        for websocket, subscriptions in self.client_subscriptions.items():
            if channel in subscriptions or "global" in subscriptions:
                try:
                    await websocket.send_json(message)
                except:
                    # Connection closed, will be cleaned up
                    pass

    async def send_to_job(self, job_id: str, message: dict):
        """Send message to all clients following a specific job"""
        await self.broadcast(message, channel=f"job:{job_id}")

    def subscribe(self, websocket: WebSocket, channel: str):
        """Subscribe client to a channel"""
        if websocket in self.client_subscriptions:
            self.client_subscriptions[websocket].add(channel)

manager = ConnectionManager()
```

### WebSocket Router

```python
from fastapi import APIRouter, WebSocket, WebSocketDisconnect, Depends, status
from .dependencies import verify_websocket_token
import uuid

router = APIRouter()

@router.websocket("/ws/{client_id}")
async def websocket_endpoint(
    websocket: WebSocket,
    client_id: str,
    token: str = Depends(verify_websocket_token)
):
    await manager.connect(websocket, client_id)
    try:
        while True:
            # Receive messages from client
            data = await websocket.receive_json()
            message_type = data.get("type")

            if message_type == "subscribe":
                # Subscribe to channels (e.g., job updates)
                channel = data.get("channel")
                manager.subscribe(websocket, channel)
                await manager.send_personal_message(
                    {"type": "subscribed", "channel": channel},
                    websocket
                )

            elif message_type == "ping":
                # Heartbeat response
                await manager.send_personal_message(
                    {"type": "pong", "timestamp": time.time()},
                    websocket
                )

            elif message_type == "unsubscribe":
                channel = data.get("channel")
                if websocket in manager.client_subscriptions:
                    manager.client_subscriptions[websocket].discard(channel)
                await manager.send_personal_message(
                    {"type": "unsubscribed", "channel": channel},
                    websocket
                )

    except WebSocketDisconnect:
        manager.disconnect(websocket, client_id)
    except Exception as e:
        logging.error(f"WebSocket error: {e}")
        manager.disconnect(websocket, client_id)
```

### Authentication

```python
from fastapi import Depends, HTTPException, status
from jose import jwt, JWTError

async def verify_websocket_token(websocket: WebSocket) -> str:
    """Verify JWT token from WebSocket query params"""
    token = websocket.query_params.get("token")
    if not token:
        await websocket.close(code=status.WS_1008_POLICY_VIOLATION)
        raise HTTPException(status_code=401, detail="Token missing")

    try:
        payload = jwt.decode(token, SECRET_KEY, algorithms=["HS256"])
        return payload.get("sub")
    except JWTError:
        await websocket.close(code=status.WS_1008_POLICY_VIOLATION)
        raise HTTPException(status_code=401, detail="Invalid token")
```

### Heartbeat/Keep-Alive

```python
async def heartbeat_task(websocket: WebSocket, interval: int = 30):
    """Send periodic heartbeat to keep connection alive"""
    try:
        while True:
            await asyncio.sleep(interval)
            await websocket.send_json({
                "type": "heartbeat",
                "timestamp": datetime.utcnow().isoformat()
            })
    except:
        pass  # Connection closed

# Start heartbeat when connection established
asyncio.create_task(heartbeat_task(websocket))
```

### Client-Side Example (TypeScript)

```typescript
class WebSocketClient {
  private ws: WebSocket | null = null;
  private reconnectAttempts = 0;
  private maxReconnectAttempts = 5;

  connect(token: string, clientId: string) {
    this.ws = new WebSocket(`ws://localhost:8000/api/v1/ws/${clientId}?token=${token}`);

    this.ws.onopen = () => {
      console.log('WebSocket connected');
      this.reconnectAttempts = 0;
    };

    this.ws.onmessage = (event) => {
      const message = JSON.parse(event.data);
      this.handleMessage(message);
    };

    this.ws.onclose = () => {
      console.log('WebSocket closed');
      this.reconnect(token, clientId);
    };

    this.ws.onerror = (error) => {
      console.error('WebSocket error:', error);
    };
  }

  private reconnect(token: string, clientId: string) {
    if (this.reconnectAttempts < this.maxReconnectAttempts) {
      this.reconnectAttempts++;
      setTimeout(() => {
        console.log(`Reconnecting (attempt ${this.reconnectAttempts})...`);
        this.connect(token, clientId);
      }, 1000 * this.reconnectAttempts);
    }
  }

  subscribe(channel: string) {
    this.send({ type: 'subscribe', channel });
  }

  send(message: any) {
    if (this.ws?.readyState === WebSocket.OPEN) {
      this.ws.send(JSON.stringify(message));
    }
  }

  private handleMessage(message: any) {
    switch (message.type) {
      case 'progress':
        // Update progress bar
        break;
      case 'error':
        // Show error notification
        break;
      case 'complete':
        // Show completion message
        break;
      case 'heartbeat':
        // Connection alive
        break;
    }
  }
}
```

### Integration Adapter Pattern

```python
# file_organizer/integrations/base.py

from abc import ABC, abstractmethod
from typing import Any, Dict, Optional
from dataclasses import dataclass
from enum import Enum

class IntegrationType(Enum):
    """Integration types."""
    DESKTOP_APP = "desktop_app"
    EDITOR = "editor"
    WORKFLOW = "workflow"
    BROWSER = "browser"
    API = "api"

@dataclass
class IntegrationConfig:
    """Integration configuration."""
    name: str
    type: IntegrationType
    enabled: bool
    auth_method: str  # "api_key", "oauth", "none"
    settings: Dict[str, Any]

class Integration(ABC):
    """Base integration interface."""

    def __init__(self, config: IntegrationConfig):
        self.config = config
        self.connected = False

    @abstractmethod
    async def connect(self) -> bool:
        """Connect to the integration."""
        pass

    @abstractmethod
    async def disconnect(self) -> None:
        """Disconnect from the integration."""
        pass

    @abstractmethod
    async def validate_auth(self) -> bool:
        """Validate authentication."""
        pass

    @abstractmethod
    async def send_file(self, file_path: str, metadata: Dict[str, Any]) -> bool:
        """Send file to integration."""
        pass

    @abstractmethod
    async def get_status(self) -> Dict[str, Any]:
        """Get integration status."""
        pass
```

### Integration Manager

```python
# file_organizer/integrations/manager.py

from typing import Dict, List, Optional
from pathlib import Path

class IntegrationManager:
    """Manages all integrations."""

    def __init__(self, config_dir: Path):
        self.config_dir = config_dir
        self._integrations: Dict[str, Integration] = {}

    def register_integration(self, integration: Integration) -> None:
        """Register an integration."""
        self._integrations[integration.config.name] = integration

    def unregister_integration(self, name: str) -> None:
        """Unregister an integration."""
        if name in self._integrations:
            del self._integrations[name]

    async def connect_all(self) -> None:
        """Connect all enabled integrations."""
        for integration in self._integrations.values():
            if integration.config.enabled:
                await integration.connect()

    async def disconnect_all(self) -> None:
        """Disconnect all integrations."""
        for integration in self._integrations.values():
            if integration.connected:
                await integration.disconnect()

    def get_integration(self, name: str) -> Optional[Integration]:
        """Get integration by name."""
        return self._integrations.get(name)

    def list_integrations(self) -> List[IntegrationConfig]:
        """List all integrations."""
        return [i.config for i in self._integrations.values()]
```

### Obsidian Integration

```python
# file_organizer/integrations/obsidian.py

import requests
from pathlib import Path
from typing import Dict, Any

class ObsidianIntegration(Integration):
    """Obsidian vault integration."""

    def __init__(self, config: IntegrationConfig):
        super().__init__(config)
        self.vault_path = Path(config.settings.get("vault_path", ""))
        self.api_url = config.settings.get("api_url", "http://localhost:27123")

    async def connect(self) -> bool:
        """Connect to Obsidian vault."""
        try:
            # Check if vault exists
            if not self.vault_path.exists():
                return False

            # Verify Obsidian Local REST API is available
            response = requests.get(f"{self.api_url}/vault/")
            self.connected = response.status_code == 200
            return self.connected
        except Exception as e:
            print(f"Failed to connect to Obsidian: {e}")
            return False

    async def disconnect(self) -> None:
        """Disconnect from Obsidian."""
        self.connected = False

    async def validate_auth(self) -> bool:
        """Validate authentication."""
        # Obsidian Local REST API uses API key
        api_key = self.config.settings.get("api_key")
        if not api_key:
            return False

        try:
            headers = {"Authorization": f"Bearer {api_key}"}
            response = requests.get(f"{self.api_url}/vault/", headers=headers)
            return response.status_code == 200
        except Exception:
            return False

    async def send_file(self, file_path: str, metadata: Dict[str, Any]) -> bool:
        """Send file to Obsidian vault."""
        try:
            source = Path(file_path)
            dest = self.vault_path / source.name

            # Copy file to vault
            import shutil
            shutil.copy2(source, dest)

            # Create note with metadata
            note_content = self._create_note_content(metadata)
            note_path = dest.with_suffix(".md")
            note_path.write_text(note_content)

            return True
        except Exception as e:
            print(f"Failed to send file to Obsidian: {e}")
            return False

    async def get_status(self) -> Dict[str, Any]:
        """Get Obsidian integration status."""
        return {
            "connected": self.connected,
            "vault_path": str(self.vault_path),
            "api_available": self.connected
        }

    def _create_note_content(self, metadata: Dict[str, Any]) -> str:
        """Create Obsidian note from metadata."""
        lines = ["---"]
        for key, value in metadata.items():
            lines.append(f"{key}: {value}")
        lines.append("---")
        lines.append("")
        lines.append(f"# {metadata.get('title', 'Untitled')}")
        return "
".join(lines)
```

```python
    """VS Code extension integration."""

        super().__init__(config)
        self.extension_id = config.settings.get("extension_id", "file-organizer.vscode")

    async def connect(self) -> bool:
        """Connect to VS Code."""
        try:
            # Check if VS Code is installed
            result = subprocess.run(
                ["code", "--version"],
                capture_output=True,
                text=True
            )
            self.connected = result.returncode == 0
            return self.connected
        except FileNotFoundError:
            return False

    async def disconnect(self) -> None:
        """Disconnect from VS Code."""
        self.connected = False

    async def validate_auth(self) -> bool:
        """Validate authentication."""
        # VS Code extension uses local socket, no auth needed
        return self.connected

    async def send_file(self, file_path: str, metadata: Dict[str, Any]) -> bool:
        """Open file in VS Code."""
        try:
            subprocess.run(["code", file_path], check=True)
            return True
        except subprocess.CalledProcessError:
            return False

    async def get_status(self) -> Dict[str, Any]:
        """Get VS Code integration status."""
        return {
            "connected": self.connected,
            "extension_installed": await self._check_extension_installed()
        }

    async def _check_extension_installed(self) -> bool:
        """Check if File Organizer extension is installed."""

            result = subprocess.run(
                ["code", "--list-extensions"],
                capture_output=True,
                text=True

            return False
```

### Browser Extension (JavaScript)

```javascript
// browser-extension/background.js

class FileOrganizerExtension {
  constructor() {
    this.apiUrl = 'http://localhost:8000';
    this.token = null;
  }

  async initialize() {
    // Load token from storage
    const result = await chrome.storage.local.get(['apiToken']);
    this.token = result.apiToken;

    // Listen for downloads
    chrome.downloads.onCreated.addListener((download) => {
      this.handleDownload(download);
    });
  }

  async handleDownload(download) {
    if (!this.token) {
      console.log('No API token configured');
      return;
    }

    // Organize downloaded file
    try {
      const response = await fetch(`${this.apiUrl}/api/browser/organize`, {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          'X-Extension-Token': this.token
        },
        body: JSON.stringify({
          url: download.url,
          filename: download.filename
        })
      });

      const result = await response.json();
      console.log('File organized:', result);

      // Show notification
      chrome.notifications.create({
        type: 'basic',
        iconUrl: 'icon.png',
        title: 'File Organized',
        message: `Organized to: ${result.organized_path}`
      });
    } catch (error) {
      console.error('Failed to organize file:', error);
    }
  }

  async getConfig() {
    const response = await fetch(`${this.apiUrl}/api/browser/config`, {
      headers: {
        'X-Extension-Token': this.token
      }
    });
    return await response.json();
  }
}

// Initialize extension
const extension = new FileOrganizerExtension();
extension.initialize();
```
